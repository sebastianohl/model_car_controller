"<html>\n"
"<header>\n"
"<script lang=\"javascript\">\n"
"    getData(\"servo1\");\n"
"    getData(\"servo2\");\n"
"    function getData(servo) {\n"
"        var xhttp = new XMLHttpRequest();\n"
"        xhttp.onreadystatechange = function () {\n"
"            if (this.readyState == 4 && this.status == 200) {\n"
"                document.getElementById(servo).value =\n"
"                    this.responseText;\n"
"                updateDisplay();\n"
"            }\n"
"        };\n"
"        xhttp.open(\"GET\", \"read_\" + servo, true);\n"
"        xhttp.send();\n"
"    }\n"
"    function setData() {\n"
"        var xhttp = new XMLHttpRequest();\n"
"        xhttp.open(\"GET\", \"save?servo1_factor=\" + document.getElementById(\"servo1\").value+\"&servo2_factor=\"+document.getElementById(\"servo2\").value, true);\n"
"        xhttp.send();\n"
"    }\n"
"    function updateDisplay() {\n"
"        document.getElementById(\"servo1_factor\").innerHTML = document.getElementById(\"servo1\").value; \n"
"        document.getElementById(\"servo2_factor\").innerHTML = document.getElementById(\"servo2\").value; \n"
"    }\n"
"</script>\n"
"</header>\n"
